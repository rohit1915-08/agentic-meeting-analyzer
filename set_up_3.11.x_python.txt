# 1. Deactivate the current environment
deactivate

# 2. Delete the old venv folder
Remove-Item -Recurse -Force venv

# 3. Create a new venv specifying Python 3.11 (Requires Python 3.11 to be installed on your system)
py -3.11 -m venv venv

# 4. Activate the new environment
.\venv\Scripts\activate

# 5. Verify the version
python --version

#Note: If the py -3.11 command fails, it means Python 3.11 is not installed on your Windows machine. 
#You will need to download and install the Windows installer for Python 3.11 from python.org first 
#(make sure to check "Add Python to PATH" during installation).
